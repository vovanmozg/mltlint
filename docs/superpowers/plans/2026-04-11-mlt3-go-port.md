# mlt3 Go Port Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port the mlt2 Ruby media-duplicate-finder to Go as a statically-linked binary with no external C dependencies.

**Architecture:** Two-phase pipeline (phase 1: scan+hash originals → build immutable indexes; phase 2: scan+hash unsorted → classify against indexes). Packages: `internal/cache` (bbolt), `internal/scanner` (FS walk, MD5, phash), `internal/comparator` (classify logic), `internal/report` (JSON + shell formatters), `cmd/mltlint` (CLI orchestration). Strict TDD: red-green-refactor per requirement.

**Tech Stack:** Go 1.22+, `github.com/corona10/goimagehash`, `go.etcd.io/bbolt`, `github.com/stretchr/testify`, `golang.org/x/image`

**Design spec:** `docs/superpowers/specs/2026-04-09-mlt3-go-port-design.md`

package cache

import (
	"encoding/binary"
	"fmt"

	bolt "go.etcd.io/bbolt"
)

var bucketName = []byte("phash-v1")

// Cache is a bbolt-backed phash cache.
type Cache struct {
	db *bolt.DB
}

// Open opens (or creates) a bbolt database at path.
func Open(path string) (*Cache, error) {
	db, err := bolt.Open(path, 0600, nil)
	if err != nil {
		return nil, fmt.Errorf("cache: open %s: %w", path, err)
	}
	err = db.Update(func(tx *bolt.Tx) error {
		_, err := tx.CreateBucketIfNotExists(bucketName)
		return err
	})
	if err != nil {
		db.Close()
		return nil, fmt.Errorf("cache: create bucket: %w", err)
	}
	return &Cache{db: db}, nil
}

// Close closes the database.
func (c *Cache) Close() error {
	return c.db.Close()
}

// cacheKey encodes path+mtime+size into a composite key.
func cacheKey(path string, mtime, size int64) []byte {
	b := make([]byte, len(path)+16)
	copy(b, path)
	binary.BigEndian.PutUint64(b[len(path):], uint64(mtime))
	binary.BigEndian.PutUint64(b[len(path)+8:], uint64(size))
	return b
}

// Get returns the cached phash for (path, mtime, size).
// Returns (0, false) if no entry exists or the entry is stale.
func (c *Cache) Get(path string, mtime, size int64) (uint64, bool) {
	var hash uint64
	var found bool
	c.db.View(func(tx *bolt.Tx) error {
		b := tx.Bucket(bucketName)
		v := b.Get(cacheKey(path, mtime, size))
		if v != nil && len(v) == 8 {
			hash = binary.BigEndian.Uint64(v)
			found = true
		}
		return nil
	})
	return hash, found
}
```

package scanner

import (
	"context"
	"net/http"
	"os"
	"path/filepath"
	"strings"

	"github.com/vp/mlt3/internal/cache"
)

// Supported image extensions.
var mediaExts = map[string]bool{
	".jpg": true, ".jpeg": true, ".png": true, ".gif": true,
	".bmp": true, ".tif": true, ".tiff": true, ".webp": true,
}

// FileInfo holds metadata for a scanned media file.
type FileInfo struct {
	Path         string
	RelativePath string
	Size         int64
	Mtime        int64
	Width        int
	Height       int
	MD5          string
	PHash        uint64
	Err          error
}

// Scanner walks directories and computes hashes.
type Scanner struct {
	cache   *cache.Cache
	workers int
	debug   bool
}

// New creates a Scanner. cache may be nil (no caching).
func New(c *cache.Cache, workers int, debug bool) *Scanner {
	if workers < 1 {
		workers = 1
	}
	return &Scanner{cache: c, workers: workers, debug: debug}
}

// Scan walks dir, finds media files, and returns their metadata.
// Does NOT compute MD5 or phash.
func (s *Scanner) Scan(ctx context.Context, dir string) ([]*FileInfo, error) {
	var files []*FileInfo
	err := filepath.Walk(dir, func(path string, info os.FileInfo, err error) error {
		if err != nil {
			return nil // skip inaccessible entries
		}
		if ctx.Err() != nil {
			return ctx.Err()
		}
		if info.IsDir() {
			return nil
		}
		if info.Size() == 0 {
			return nil // SCAN.SKIP.EMPTY
		}
		ext := strings.ToLower(filepath.Ext(path))
		if !mediaExts[ext] {
			return nil
		}

		// Check content type
		if !isMediaContent(path) {
			return nil // SCAN.SKIP
		}

		rel, _ := filepath.Rel(dir, path)
		fi := &FileInfo{
			Path:         path,
			RelativePath: rel,
			Size:         info.Size(),
			Mtime:        info.ModTime().Unix(),
		}
		files = append(files, fi)
		return nil
	})
	return files, err
}

// isMediaContent reads the first 512 bytes and checks if the content is an image.
func isMediaContent(path string) bool {
	f, err := os.Open(path)
	if err != nil {
		return false
	}
	defer f.Close()
	buf := make([]byte, 512)
	n, _ := f.Read(buf)
	if n == 0 {
		return false
	}
	ct := http.DetectContentType(buf[:n])
	return strings.HasPrefix(ct, "image/")
}
```

import (
	"crypto/md5"
	"encoding/hex"
	"io"
	"sync"

// ComputeMD5 computes MD5 for files that don't have it yet, in parallel.
func (s *Scanner) ComputeMD5(ctx context.Context, files []*FileInfo) error {
	var wg sync.WaitGroup
	sem := make(chan struct{}, s.workers)

	for _, f := range files {
		if f.MD5 != "" || f.Err != nil {
			continue
		}
		wg.Add(1)
		sem <- struct{}{}
		go func(fi *FileInfo) {
			defer wg.Done()
			defer func() { <-sem }()
			if ctx.Err() != nil {
				return
			}
			h, err := md5File(fi.Path)
			if err != nil {
				fi.Err = err
				return
			}
			fi.MD5 = h
		}(f)
	}
	wg.Wait()
	return ctx.Err()
}

func md5File(path string) (string, error) {
	f, err := os.Open(path)
	if err != nil {
		return "", err
	}
	defer f.Close()
	h := md5.New()
	if _, err := io.Copy(h, f); err != nil {
		return "", err
	}
	return hex.EncodeToString(h.Sum(nil)), nil
}

import (
	"github.com/corona10/goimagehash"

// ComputePHash computes perceptual hashes for files that don't have one yet.
// Consults the cache if available.
func (s *Scanner) ComputePHash(ctx context.Context, files []*FileInfo) error {
	var wg sync.WaitGroup
	sem := make(chan struct{}, s.workers)

	for _, f := range files {
		if f.PHash != 0 || f.Err != nil {
			continue
		}
		wg.Add(1)
		sem <- struct{}{}
		go func(fi *FileInfo) {
			defer wg.Done()
			defer func() { <-sem }()
			if ctx.Err() != nil {
				return
			}

			// Check cache
			if s.cache != nil {
				if h, ok := s.cache.Get(fi.Path, fi.Mtime, fi.Size); ok {
					fi.PHash = h
					return
				}
			}

			h, err := phashFile(fi.Path)
			if err != nil {
				fi.Err = err
				return
			}
			fi.PHash = h

			// Store in cache
			if s.cache != nil {
				_ = s.cache.Set(fi.Path, fi.Mtime, fi.Size, h)
			}
		}(f)
	}
	wg.Wait()
	return ctx.Err()
}

func phashFile(path string) (uint64, error) {
	f, err := os.Open(path)
	if err != nil {
		return 0, err
	}
	defer f.Close()
	img, _, err := image.Decode(f)
	if err != nil {
		return 0, err
	}
	hash, err := goimagehash.PerceptionHash(img)
	if err != nil {
		return 0, err
	}
	return hash.GetHash(), nil
}

package comparator

import (
	"context"
	"math"
	"sort"

	"github.com/vp/mlt3/internal/scanner"
)

const doubtfulThreshold = 2

// Result holds the classification of one unsorted file.
type Result struct {
	Duplicate *scanner.FileInfo
	Original  *scanner.FileInfo
	Level     string  // "similar" | "doubtful"
	Distance  int
	Score     float64
}

// Indexes holds read-only lookup structures built from originals.
type Indexes struct {
	BySize  map[int64][]*scanner.FileInfo
	ByRatio map[float64][]*scanner.FileInfo
}

// HashComputer is the interface for on-demand hash computation.
type HashComputer interface {
	ComputeMD5(ctx context.Context, files []*scanner.FileInfo) error
	ComputePHash(ctx context.Context, files []*scanner.FileInfo) error
}

// Ratio computes the rounded aspect ratio for a FileInfo.
func Ratio(f *scanner.FileInfo) float64 {
	if f.Width == 0 || f.Height == 0 {
		return 0
	}
	return math.Round(float64(f.Width)/float64(f.Height)*10) / 10
}

// BuildIndexes creates read-only indexes from a fully hashed originals slice.
func BuildIndexes(originals []*scanner.FileInfo) *Indexes {
	idx := &Indexes{
		BySize:  make(map[int64][]*scanner.FileInfo),
		ByRatio: make(map[float64][]*scanner.FileInfo),
	}
	// Sort originals by path for stable ordering
	sort.Slice(originals, func(i, j int) bool {
		return originals[i].Path < originals[j].Path
	})
	for _, f := range originals {
		if f.Err != nil {
			continue
		}
		idx.BySize[f.Size] = append(idx.BySize[f.Size], f)
		r := Ratio(f)
		if r > 0 {
			idx.ByRatio[r] = append(idx.ByRatio[r], f)
		}
	}
	return idx
}

// Classify determines if an unsorted file is a duplicate of any original.
// Returns nil if no match found.
func Classify(
	ctx context.Context,
	file *scanner.FileInfo,
	indexes *Indexes,
	hc HashComputer,
) (*Result, error) {
	if file.Err != nil {
		return nil, nil
	}

	// Step 1: MD5 fast path
	if candidates, ok := indexes.BySize[file.Size]; ok && len(candidates) > 0 {
		// Ensure file has MD5
		if file.MD5 == "" {
			if err := hc.ComputeMD5(ctx, []*scanner.FileInfo{file}); err != nil {
				return nil, err
			}
		}
		// Ensure candidates have MD5
		var needMD5 []*scanner.FileInfo
		for _, c := range candidates {
			if c.MD5 == "" {
				needMD5 = append(needMD5, c)
			}
		}
		if len(needMD5) > 0 {
			if err := hc.ComputeMD5(ctx, needMD5); err != nil {
				return nil, err
			}
		}
		// Check for MD5 match
		if file.MD5 != "" {
			for _, c := range candidates {
				if c.MD5 == file.MD5 {
					return &Result{
						Duplicate: file,
						Original:  c,
						Level:     "similar",
						Distance:  0,
						Score:     1.0,
					}, nil
				}
			}
		}
	}

	// Step 2: phash path
	if file.PHash == 0 {
		if err := hc.ComputePHash(ctx, []*scanner.FileInfo{file}); err != nil {
			return nil, err
		}
	}
	if file.PHash == 0 {
		return nil, nil // could not compute phash
	}

	fileRatio := Ratio(file)
	if fileRatio == 0 {
		return nil, nil // no dimensions, can't compare by phash
	}

	candidates, ok := indexes.ByRatio[fileRatio]
	if !ok || len(candidates) == 0 {
		return nil, nil
	}

	var bestOrig *scanner.FileInfo
	bestDist := doubtfulThreshold + 1

	for _, c := range candidates {
		if c.PHash == 0 {
			continue
		}
		d := hammingDistance(file.PHash, c.PHash)
		if d < bestDist {
			bestDist = d
			bestOrig = c
		}
	}

	if bestOrig == nil || bestDist > doubtfulThreshold {
		return nil, nil
	}

	level := "doubtful"
	score := 1.0 - float64(bestDist)/3.0
	score = math.Round(score*100) / 100

	if bestDist == 0 {
		level = "similar"
		score = 1.0
	}

	return &Result{
		Duplicate: file,
		Original:  bestOrig,
		Level:     level,
		Distance:  bestDist,
		Score:     score,
	}, nil
}

func hammingDistance(a, b uint64) int {
	x := a ^ b
	count := 0
	for x != 0 {
		count++
		x &= x - 1
	}
	return count
}

import (
	"encoding/json"

	"github.com/vp/mlt3/internal/scanner"

const version = "1.0.0"

// WriteJSON writes the rmlint-compatible JSON report.
func WriteJSON(w io.Writer, cfg Config, groups []*Group, stats Stats) error {
	var records []interface{}

	// Header
	records = append(records, map[string]interface{}{
		"description":   "mlt3 duplicate report",
		"cwd":           cfg.Cwd,
		"args":          cfg.Args,
		"version":       version,
		"rev":           "",
		"progress":      0,
		"checksum_type": "phash-dct64",
		"type":          "header",
	})

	// Duplicate file records
	totalDups := 0
	totalLintSize := int64(0)
	for _, g := range groups {
		twins := 1 + len(g.Duplicates) // original + duplicates
		// Original record
		records = append(records, fileRecord(g.Original, twins, true, stats.Progress, nil, "", cfg))
		// Duplicate records
		for _, r := range g.Duplicates {
			records = append(records, fileRecord(r.Duplicate, twins, false, stats.Progress, r, cfg.Dups, cfg))
			totalDups++
			totalLintSize += r.Duplicate.Size
		}
	}

	// Footer
	records = append(records, map[string]interface{}{
		"aborted":         stats.Aborted,
		"progress":        stats.Progress,
		"total_files":     stats.TotalFiles,
		"ignored_files":   stats.IgnoredFiles,
		"ignored_folders": stats.IgnoredFolders,
		"duplicates":      totalDups,
		"duplicate_sets":  len(groups),
		"total_lint_size": totalLintSize,
		"type":            "footer",
	})

	enc := json.NewEncoder(w)
	enc.SetIndent("", "  ")
	return enc.Encode(records)
}

func fileRecord(fi *scanner.FileInfo, twins int, isOriginal bool, progress int, result *Result, dupsDir string, cfg Config) map[string]interface{} {
	rec := map[string]interface{}{
		"id":          crc32.ChecksumIEEE([]byte(fi.Path)),
		"type":        "duplicate_file",
		"progress":    progress,
		"checksum":    fi.MD5,
		"path":        fi.Path,
		"size":        fi.Size,
		"twins":       twins,
		"depth":       strings.Count(fi.RelativePath, "/") + 1,
		"inode":       fileInode(fi.Path),
		"disk_id":     fileDiskID(fi.Path),
		"is_original": isOriginal,
		"mtime":       fi.Mtime,
	}
	if !isOriginal && result != nil {
		rec["similarity"] = map[string]interface{}{
			"level":    result.Level,
			"distance": result.Distance,
			"score":    result.Score,
		}
		rec["move_to"] = fmt.Sprintf("%s/%s/%s", dupsDir, result.Level, result.Duplicate.RelativePath)
	}
	return rec
}

// Result re-exported for use in fileRecord. Wraps comparator.Result fields needed here.
type Result = comparator.Result

func fileInode(path string) uint64 {
	var stat syscall.Stat_t
	if err := syscall.Stat(path, &stat); err != nil {
		return 0
	}
	return stat.Ino
}

func fileDiskID(path string) int32 {
	var stat syscall.Stat_t
	if err := syscall.Stat(path, &stat); err != nil {
		return 0
	}
	return stat.Dev
}

// WriteShell writes the rmlint-style shell script with move_cmd.
func WriteShell(w io.Writer, cfg Config, groups []*Group) error {
	// Header
	fmt.Fprintf(w, "#!/usr/bin/env bash\n")
	fmt.Fprintf(w, "# This file was autogenerated by mlt3 mltlint\n")
	fmt.Fprintf(w, "# Your command line was: %s\n\n", cfg.Args)
	fmt.Fprintf(w, "DUPS_DIR=%q\n\n", cfg.Dups)

	// Function definitions
	fmt.Fprintf(w, "handle_emptydir() { :; }\n")
	fmt.Fprintf(w, "handle_emptyfile() { :; }\n")
	fmt.Fprintf(w, "original_cmd() { :; }\n\n")

	fmt.Fprintf(w, `move_cmd() {
  local duplicate="$1"
  local original="$2"
  local dest="$3"
  if [ ! -e "$original" ]; then
    echo "SKIP (original missing): $original"
    return
  fi
  local dest_dir
  dest_dir="$(dirname "$dest")"
  mkdir -p "$dest_dir"
  mv "$duplicate" "$dest"
}
`)
	fmt.Fprintf(w, "\n######### START OF AUTOGENERATED OUTPUT #########\n\n")

	// Groups
	for i, g := range groups {
		if i > 0 {
			fmt.Fprintf(w, "\n")
		}
		fmt.Fprintf(w, "original_cmd '%s' # original\n", g.Original.Path)
		for _, r := range g.Duplicates {
			dest := fmt.Sprintf("%s/%s/%s", cfg.Dups, r.Level, r.Duplicate.RelativePath)
			fmt.Fprintf(w, "move_cmd '%s' '%s' '%s' # %s (dist=%d)\n",
				r.Duplicate.Path, g.Original.Path, dest, r.Level, r.Distance)
		}
	}

	fmt.Fprintf(w, "\n######### END OF AUTOGENERATED OUTPUT #########\n")
	return nil
}

package report

import (
	"fmt"
	"io"
)

// WriteConsoleSummary writes the rmlint-style summary to stderr.
func WriteConsoleSummary(w io.Writer, cfg Config, groups []*Group, stats Stats) error {
	// Duplicate listing
	fmt.Fprintf(w, "\n# Duplicate(s):\n")
	for _, g := range groups {
		fmt.Fprintf(w, "    original '%s'\n", g.Original.Path)
		for _, r := range g.Duplicates {
			fmt.Fprintf(w, "    %-8s '%s' (dist=%d)\n", r.Level, r.Duplicate.Path, r.Distance)
		}
		fmt.Fprintf(w, "\n")
	}

	// Summary
	totalDups := 0
	var totalLintSize int64
	for _, g := range groups {
		totalDups += len(g.Duplicates)
		for _, r := range g.Duplicates {
			totalLintSize += r.Duplicate.Size
		}
	}

	fmt.Fprintf(w, "==> Note: Please use the saved script below for moving, not the above output.\n")
	fmt.Fprintf(w, "==> In total %d files, whereof %d are duplicates in %d groups.\n",
		stats.TotalFiles, totalDups, len(groups))
	fmt.Fprintf(w, "==> This equals %s of duplicates which could be moved.\n",
		humanBytes(totalLintSize))
	fmt.Fprintf(w, "==> Scanning took in total %.1fs.\n", stats.Elapsed.Seconds())

	return nil
}

func humanBytes(b int64) string {
	const unit = 1024
	if b < unit {
		return fmt.Sprintf("%d B", b)
	}
	div, exp := int64(unit), 0
	for n := b / unit; n >= unit; n /= unit {
		div *= unit
		exp++
	}
	return fmt.Sprintf("%.2f %cB", float64(b)/float64(div), "KMGTPE"[exp])
}

	// Validate directories exist
	for _, d := range []struct{ name, path string }{
		{"originals", cfg.Originals},
		{"unsorted", cfg.Unsorted},
	} {
		info, err := os.Stat(d.path)
		if err != nil || !info.IsDir() {
			fmt.Fprintf(os.Stderr, "Error: --%s=%s is not a readable directory\n", d.name, d.path)
			os.Exit(1)
		}
	}

	return cfg
}

func defaultCachePath() string {
	dir := os.Getenv("XDG_CACHE_HOME")
	if dir == "" {
		home, _ := os.UserHomeDir()
		dir = filepath.Join(home, ".cache")
	}
	return filepath.Join(dir, "mltlint", "cache.db")
}

func run(cfg config) int {
	start := time.Now()

	ctx, cancel := signal.NotifyContext(context.Background(), os.Interrupt, syscall.SIGTERM)
	defer cancel()

	// Open cache
	if err := os.MkdirAll(filepath.Dir(cfg.CachePath), 0755); err != nil {
		fmt.Fprintf(os.Stderr, "Error: cannot create cache dir: %v\n", err)
		return 1
	}
	c, err := cache.Open(cfg.CachePath)
	if err != nil {
		fmt.Fprintf(os.Stderr, "Error: cannot open cache: %v\n", err)
		return 1
	}
	defer c.Close()

	sc := scanner.New(c, cfg.Workers, cfg.Debug)

	// Phase 1: originals
	fmt.Fprintf(os.Stderr, "[phase 1/2] Scanning originals...\n")
	origFiles, err := sc.Scan(ctx, cfg.Originals)
	if err != nil {
		fmt.Fprintf(os.Stderr, "Error scanning originals: %v\n", err)
		return 1
	}
	fmt.Fprintf(os.Stderr, "[phase 1/2] originals: scanned %d, computing phash...\n", len(origFiles))

	if err := sc.ComputePHash(ctx, origFiles); err != nil {
		if ctx.Err() != nil {
			fmt.Fprintf(os.Stderr, "\nInterrupted during phase 1. No results to write.\n")
			return 130
		}
		fmt.Fprintf(os.Stderr, "Error computing phash for originals: %v\n", err)
		return 1
	}
	fmt.Fprintf(os.Stderr, "[phase 1/2] originals: indexed %d files\n", len(origFiles))

	indexes := comparator.BuildIndexes(origFiles)

	// Phase 2: unsorted
	fmt.Fprintf(os.Stderr, "[phase 2/2] Scanning unsorted...\n")
	unsortedFiles, err := sc.Scan(ctx, cfg.Unsorted)
	if err != nil {
		if ctx.Err() != nil {
			fmt.Fprintf(os.Stderr, "\nInterrupted during phase 2 scan. No results to write.\n")
			return 130
		}
		fmt.Fprintf(os.Stderr, "Error scanning unsorted: %v\n", err)
		return 1
	}
	fmt.Fprintf(os.Stderr, "[phase 2/2] unsorted: scanned %d, computing phash...\n", len(unsortedFiles))

	if err := sc.ComputePHash(ctx, unsortedFiles); err != nil && ctx.Err() == nil {
		fmt.Fprintf(os.Stderr, "Error computing phash for unsorted: %v\n", err)
		return 1
	}

	// Classify
	aborted := ctx.Err() != nil
	groups := classifyAndGroup(ctx, unsortedFiles, indexes, sc)

	if ctx.Err() != nil {
		aborted = true
	}

	// Stats
	ignoredFiles := 0
	for _, f := range origFiles {
		if f.Err != nil {
			ignoredFiles++
		}
	}
	for _, f := range unsortedFiles {
		if f.Err != nil {
			ignoredFiles++
		}
	}

	classified := 0
	for _, g := range groups {
		classified += len(g.Duplicates)
	}

	progress := 100
	if aborted && len(unsortedFiles) > 0 {
		progress = classified * 100 / len(unsortedFiles)
	}

	stats := report.Stats{
		TotalFiles:   len(origFiles) + len(unsortedFiles),
		IgnoredFiles: ignoredFiles,
		Aborted:      aborted,
		Progress:     progress,
		Elapsed:      time.Since(start),
	}

	// Write outputs
	rcfg := report.Config{
		Args:      fmt.Sprintf("mltlint --originals=%s --unsorted=%s --dups=%s", cfg.Originals, cfg.Unsorted, cfg.Dups),
		Cwd:       cfg.Output,
		Originals: cfg.Originals,
		Unsorted:  cfg.Unsorted,
		Dups:      cfg.Dups,
	}

	jsonPath := filepath.Join(cfg.Output, "duplicates.json")
	shPath := filepath.Join(cfg.Output, "duplicates.sh")

	if err := atomicWrite(jsonPath, func(w *os.File) error {
		return report.WriteJSON(w, rcfg, groups, stats)
	}); err != nil {
		fmt.Fprintf(os.Stderr, "Error writing %s: %v\n", jsonPath, err)
		return 1
	}

	if err := atomicWrite(shPath, func(w *os.File) error {
		return report.WriteShell(w, rcfg, groups)
	}); err != nil {
		fmt.Fprintf(os.Stderr, "Error writing %s: %v\n", shPath, err)
		return 1
	}
	os.Chmod(shPath, 0755)

	// Console summary
	report.WriteConsoleSummary(os.Stderr, rcfg, groups, stats)
	fmt.Fprintf(os.Stderr, "\nWrote a json file to: %s\n", jsonPath)
	fmt.Fprintf(os.Stderr, "Wrote a sh file to: %s\n", shPath)

	if aborted {
		return 130
	}
	if ignoredFiles > 0 {
		return 2
	}
	return 0
}

func classifyAndGroup(ctx context.Context, files []*scanner.FileInfo, indexes *comparator.Indexes, hc comparator.HashComputer) []*report.Group {
	groupMap := make(map[string]*report.Group) // keyed by original path
	var groupOrder []string

	for _, f := range files {
		if ctx.Err() != nil {
			break
		}
		if f.Err != nil {
			continue
		}
		r, err := comparator.Classify(ctx, f, indexes, hc)
		if err != nil {
			continue
		}
		if r == nil {
			continue
		}
		key := r.Original.Path
		g, ok := groupMap[key]
		if !ok {
			g = &report.Group{Original: r.Original}
			groupMap[key] = g
			groupOrder = append(groupOrder, key)
		}
		g.Duplicates = append(g.Duplicates, r)
	}

	groups := make([]*report.Group, 0, len(groupOrder))
	for _, key := range groupOrder {
		groups = append(groups, groupMap[key])
	}
	return groups
}

func atomicWrite(path string, fn func(w *os.File) error) error {
	tmp := path + ".tmp"
	f, err := os.Create(tmp)
	if err != nil {
		return err
	}
	if err := fn(f); err != nil {
		f.Close()
		os.Remove(tmp)
		return err
	}
	if err := f.Close(); err != nil {
		os.Remove(tmp)
		return err
	}
	return os.Rename(tmp, path)
}

		}

	require.Contains(t, sh, "original_cmd")

				require.Equal(t, true, footer["aborted"])
			}
		}
	}

}
